Rating component is a star based rating input.

$('#default').puirating();
                            
<input id="rating" type="hidden" />
Name Type Default Description
stars Integer 5 Rating stars number.
cancel Boolean true Defines if user can cancel rating.
Name Parameters Description
rate event: puiratingrate event
value: Rating value. 		Fired when stars selected. 
$('#default').puirating({
    rate: function(event, value) { 
        //..
}});
Name Parameters Description
cancel - Clears selection.
option name: Name of the option Returns the value of the option.
option name: Name of the option, value: Value of the option Set the value of the option. 
$('#default').puiaccordion('cancel');  
                            
<script type="text/javascript">
        $(function() {
            $('#rating').puirating();
                
            $('#default').puirating();
                
            $('#callback').puirating({rate: function(event, value) { 
                    alert(value);
                }});
                
            $('#nocancel').puirating({cancel:false});
                
            $('#readonly').puirating({readonly:true, cancel:false});
                
            $('#disabled').puirating({disabled:true, cancel:false});
        });
    </script>
