<script type="text/javascript">
        $(function() {
            $('#in').puiinputtext();

            $('#enable').puibutton({
                click: function () {
                    $('#state').puiinputtext('enable');
                    $('#enable').hide();
                    $('#disable').show();
                }
            });

            $('#disable').puibutton({
                click: function () {
                    $('#state').puiinputtext('disable');
                    $('#enable').show();
                    $('#disable').hide();
                }
            });

            $('#enable').hide();
        });
</script>
