$("#ds").puidatascroller({
    header:'Scroll Down to Load More Cars',
    chunkSize: 10,
    loader: $('#btn'),
    datasource: function (callback) {
        $.ajax({
            type: "GET",
            url: 'showcase/resources/data/cars-large.json',
            dataType: "json",
            context: this,
            success: function (response) {
                callback.call(this, response);
            }
        });
    },
    template: $('#cartemplate')
});

$('#btn').puibutton({
    icon: 'fa-refresh'
});
