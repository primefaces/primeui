Getting Started

Basic

$('#basic').puimultiselect();
                        
<select id="basic" name="basic">
<option value="0">Istanbul</option>
<option value="1">Barcelona</option>
<option value="2">Paris</option>
<option value="3">Oslo</option>
</select>

Local Data

var themes = new Array('afterdark', 'afternoon', 'afterwork', ...);

$('#localData').puimultiselect({
data: themes
});
                        
<select id="localData" name="localData"></select>

Attributes

Name Type Default Description
defaultLabel string 'Choose' Default label of the MultiSelect.
caseSensitive false boolean Defines case sensitivity on filtering.
filterMatchMode string startsWith Mode of filter, valid values are startsWith, contains, endsWith and custom.
filterFunction function null Function to be used in custom filtering, this function gets option as parameter and needs to return a boolean.
data array null An array of items to populate dropdown dynamically. Objects in this array can be simple types or object with label-value fields like {label:'Option 1', value:1}
scrollHeight integer 200 Maximum height boundary of overlay panel in pixels, if total height of items exceed this value, scrollbar is displayed.
style string null Inline style of the element.
styleClass string null Style class of the element.
value array null Preselection value of MultiSelect.

Events

Name Parameters Description
onHide event: puimultiselectonhide event Fired on panel hide.
onShow event: puimultiselectonshow event Fired on panel show.

'cupertino', 'dark-hive', 'dot-luv', 'eggplant', 'excite-bike', 'flick', 'glass-x', 'home', 'hot-sneaks', 'humanity', 'le-frog', 'midnight',
'mint-choc', 'overcast', 'pepper-grinder', 'redmond', 'rocket', 'sam', 'smoothness', 'south-street', 'start', 'sunny', 'swanky-purse', 'trontastic',
'ui-darkness', 'ui-lightness', 'vader');

$('#basic').puimultiselect({

});

});
});
</script>
