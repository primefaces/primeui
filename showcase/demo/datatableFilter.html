DataTable - Filtering

Filtering is enabled using filter property at column configuration with customization options including filterMatchMode and custom filter function. Filtering can be done at client side or at server side in lazy mode.

  • Markup
  • Script
<input id="globalFilter" placeholder="Global Filter" size="30"></input>
<div id="tblfilter">/</div>
                        
$(function() {
    $('#tblfilter').puidatatable({
        caption: 'Filtering',
        columns: [
            {field: 'vin', headerText: 'Vin', filter: true},
            {field: 'brand', headerText: 'Brand', filter: true},
            {field: 'year', headerText: 'Year', filter: true},
            {field: 'color', headerText: 'Color', filter: true}
        ],
        globalFilter:'#globalFilter',
        datasource: function(callback) {
            $.ajax({
                type: "GET",
                url: 'showcase/resources/data/cars-large.json',
                dataType: "json",
                context: this,
                success: function(response) {
                    callback.call(this, response);
                }
            });
        },
        paginator: {
            rows: 10
        }
    });
});
