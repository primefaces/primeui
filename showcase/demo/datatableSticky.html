DataTable - Sticky Header

Header of datatable can be made sticky to be kept inside viewport during scrolling.

  • Markup
  • Script
<div id="tblsticky">/</div>
                        
$(function() {
    $('#tblsticky').puidatatable({
        caption: 'Filtering',
        columns: [
            {field: 'vin', headerText: 'Vin', filter: true},
            {field: 'brand', headerText: 'Brand', filter: true},
            {field: 'year', headerText: 'Year', filter: true},
            {field: 'color', headerText: 'Color', filter: true}
        ],
        stickyHeader: true,
        datasource: function(callback) {
            $.ajax({
                type: "GET",
                url: 'showcase/resources/data/cars-large.json',
                dataType: "json",
                context: this,
                success: function(response) {
                    callback.call(this, response);
                }
            });
        },
        paginator: {
            rows: 10
        }
    });
});
